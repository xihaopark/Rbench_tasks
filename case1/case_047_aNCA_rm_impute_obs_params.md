dir.create("outputs", showWarnings = FALSE, recursive = TRUE)

read_tsv <- function(name) {
  read.delim(file.path("inputs", name), check.names = FALSE, stringsAsFactors = FALSE)
}

true_like <- function(x) {
  trimws(tolower(as.character(x))) %in% c("true", "t", "yes", "y", "1")
}

data <- read_tsv("data.tsv")
metadata <- read_tsv("metadata_nca_parameters.tsv")
result <- data

param_cols <- intersect(c("PARAMCD", "PPTESTCD", "PARAM"), names(data))
metadata_param_cols <- intersect(c("PARAMCD", "PPTESTCD", "PARAM", "PKNCA"), names(metadata))
flag_cols <- grep("imput|obs", names(metadata), ignore.case = TRUE, value = TRUE)

if (length(param_cols) > 0 && length(metadata_param_cols) > 0 && length(flag_cols) > 0) {
  flagged <- rep(FALSE, nrow(metadata))
  for (col in flag_cols) {
    flagged <- flagged | true_like(metadata[[col]])
  }
  flagged_params <- unique(unlist(metadata[flagged, metadata_param_cols, drop = FALSE], use.names = FALSE))
  flagged_params <- flagged_params[!is.na(flagged_params) & nzchar(as.character(flagged_params))]
  if (length(flagged_params) > 0) {
    drop_row <- rep(FALSE, nrow(result))
    for (col in param_cols) {
      drop_row <- drop_row | result[[col]] %in% flagged_params
    }
    result <- result[!drop_row, , drop = FALSE]
  }
}

write.csv(result, file.path("outputs", "result.csv"), row.names = FALSE)

      across(intersect(names(df), date_cols), ~ suppressWarnings(as.Date(.x)))
    )
}

data  <- convert_types(data)
meta  <- convert_types(meta)

# Keys for joining / matching
key_vars <- c("USUBJID", "STUDYID", "PARAMCD", "PARAM", "AVISITN", "AVISIT")

# Identify observations in data that are missing required columns and can be imputed from meta
required_cols <- c("AVAL", "ADT", "BASE", "CHG", "PCHG")

data_missing <- data %>%
  mutate(
    ANY_MISS = if_any(all_of(required_cols), is.na)
  ) %>%
  filter(ANY_MISS)

# Join with metadata to obtain candidate imputation values
imputed <- data_missing %>%
  select(-ANY_MISS) %>%
  left_join(
    meta %>%
      select(all_of(c(key_vars, required_cols))),
    by = key_vars,
    suffix = c("", ".META")
  ) %>%
  mutate(
    AVAL  = if_else(is.na(AVAL),  AVAL.META,  AVAL),
    ADT   = if_else(is.na(ADT),   ADT.META,   ADT),
    BASE  = if_else(is.na(BASE),  BASE.META,  BASE),
    CHG   = if_else(is.na(CHG),   CHG.META,   CHG),
    PCHG  = if_else(is.na(PCHG),  PCHG.META,  PCHG)
  ) %>%
  select(names(data))

# Rows that did not need imputation
data_complete <- data %>%
  mutate(
    ANY_MISS = if_any(all_of(required_cols), is.na)
  ) %>%
  filter(!ANY_MISS) %>%
  select(-ANY_MISS)

# Combine imputed and complete
result <- bind_rows(
  data_complete,
  imputed
) %>%
  arrange(USUBJID, STUDYID, PARAMCD, AVISITN, AVISIT)

# Ensure only required columns in required order
required_order <- c(
  "USUBJID", "STUDYID", "PARAMCD", "PARAM",
  "AVAL", "AVISITN", "AVISIT", "ADT",
  "ANL01FL", "BASE", "CHG", "PCHG", "ABLFL"
)

result <- result %>%
  select(all_of(required_order))

# Convert ADT back to character in ISO format for output
result <- result %>%
  mutate(
    ADT = if_else(is.na(ADT), NA_character_, format(ADT, "%Y-%m-%d"))
  )

# Write result
write_csv(result, file.path("outputs", "result.csv"), na = "")
